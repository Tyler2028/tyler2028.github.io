Encapsulation Example


        //Using a class to show encapsulation
        class Person 
        {
            #name; //Private

            constructor(name) 
            {
                this.#name = name; //Private
            }

            //Public method to access to private
            getName() 
            {
                return this.#name;
            }

            //Public method to modify private attributes
            setName(newName) 
            {
                this.#name = newName;
            }
        }

        const person = new Person("Bob");
        const outputDiv = document.getElementById("output");

        //Display first name
        outputDiv.innerHTML += `<p>${person.getName()}</p>`; //Output: Bob

        //Update name and display again
        person.setName("Bill");
        outputDiv.innerHTML += `<p>${person.getName()}</p>`; //Output: Bill
