Abstraction Example

class Vehicle {
    constructor(type, wheels) {
        // Initialize the type of vehicle and number of wheels
        this.type = type;
        this.wheels = wheels;
    }

    // Method to describe the movement of the vehicle
    move() {
        return `${this.type} is moving on ${this.wheels} wheels.`;
    }
}

// Subclass representing a type of vehicle
class Car extends Vehicle {
    constructor(brand) {
        // Call the parent class constructor with 'Car' as type and 4 wheels
        super('Car', 4);
        // Initialize the brand of the car
        this.brand = brand;
    }

    // Method to show the car honking
    honk() {
        return `${this.brand} car is honking!`;
    }
}

// Example usage
// Create an instance of the Car class with the brand 'Toyota'
const myCar = new Car('Toyota');
// Log the movement of the car to the console
console.log(myCar.move());
// Log the honking action of the car to the console
console.log(myCar.honk());
