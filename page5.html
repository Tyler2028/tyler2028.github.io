Polymorphism Example

class Animal 
{
    // Method to show the sound an animal makes
    speak() 
    {
        return "This animal makes a sound.";
    }
}

// class representing Dog inheriting from Animal
{
    // Overriding the speak method to provide attubrites for Dog
    speak() 
    {
        return "The dog barks.";
    }
}

// class representing a Cat inheriting from Animal
{
    // Overriding the speak method to provide attubrites for Cat
    speak() 
    {
        return "The cat meows.";
    }
}

function demonstratePolymorphism() 
{
    // Creating an array of Animal objects, including Dog and Cat
    const animals = [new Animal(), new Dog(), new Cat()];
    
    // Using the speak method on each object, showcasing polymorphism
    // The method called depends on the actual type of the object
    return animals.map(animal => animal.speak()).join('<br>');
}
