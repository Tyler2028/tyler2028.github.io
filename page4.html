Inheritance Example

class Animal {
    constructor(name) 
    {
        // Initialize the name
        this.name = name;
    }

    // Define a method for the animal to speak
    speak()
    {
        // Return a message
        return `${this.name} makes a noise.`;
    }
}

// Define a subclass called Dog that extends the Animal class
{
    // Override the speak method for the Dog class
    speak() 
    {
        // Return a message indicating the dog barks
        return `${this.name} barks.`;
    }
}

// Example usage
// Create an instance of the Dog class with the name 'Buddy'
const dog = new Dog('Buddy');

// Display the result of the speak method in the 'output' div
document.getElementById('output').innerText = dog.speak();
